// Journal Table

create table journal(journal_id uuid primary key default gen_random_uuid(),content TEXT NOT NULL,
user_id uuid references users(id) on delete cascade,
created_at timestamp default now());

// Journal_media
create table journal_media(media_id  uuid primary key default gen_random_uuid(), url TEXT NOT NULL,
type TEXT check (type IN('image','video') ),journal_id uuid references journal(journal_id) on delete cascade);
